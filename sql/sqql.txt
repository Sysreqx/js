SELECT  COUNT(Студенты.ID), Факультеты.[ID]
  FROM
      (
        (
          Расселения INNER JOIN Студенты ON Расселения.[Код проживающего (Wiz)] = Студенты.ID
        )
          INNER JOIN Группы ON Группы.ID = Студенты.[Группа (ID группы)]
      )
      INNER JOIN Факультеты.ID = Группы.[ID факультета]
      GROUP BY Факультеты.[ID]

SELECT COUNT(Расселения.[Код проживающего (Wiz)]), Комнаты.[Номер комнаты]
FROM Расселения INNER JOIN Комнаты
ON Расселения.[Номер комнаты (Wiz)] = Комнаты.ID
GROUP BY Комнаты.[Номер комнаты];

SELECT  Комнаты.[Номер комнаты], Комнаты.[Номер эатажа], Комнаты.[Количество мест], Студенты.[Фамилия], Студенты.[Имя], Студенты.[Отчество],
        Расселения.[Проживает с даты], Расселения.[Проживает по дату],
        Студенты.[Номер зачетной книжки], Факультеты.[Название факультета], Группы.[Номер группы]
from
   (
      (
         (
            Расселения inner join Комнаты on Расселения.[Номер комнаты (Wiz)] = Комнаты.ID
         )
         inner join Студенты on Студенты.ID = Расселения.[Код проживающего (Wiz)]
      )
      inner join Группы on Группы.ID = Студенты.[Группа (ID группы)]
   )
   inner join Факультеты on Факультеты.ID = Группы.[ID факультета]
   ;
